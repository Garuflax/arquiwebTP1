\begin{frame}
    \frametitle{Posicionamiento de resultados orgánicos}
    \begin{itemize}
    \item Se basa en indexación de las arañas web (Web Crawlers)
    \item[] \begin{itemize}
            \item Estas son aplicaciones que recorren las páginas web y almacenan las palabras clave relevantes
            \item Ejemplos: Googlebot y Bingbot
            \end{itemize}
    \item SEO busca optimizar la estructura y el contenido de una página web para facilitar el procesamiento a las arañas web
    \end{itemize}
    
\end{frame}

\begin{frame}
    \frametitle{Tipos de SEO}
    \begin{itemize}
    \item Interna (On-page SEO): mejoras en el contenido, mejoras técnicas en el código, accesibilidad, etc.
    \item Externa (Off-page SEO): mejorar la notoriedad de la web mediante referencias a ella
    \end{itemize}
    
\end{frame}

\begin{frame}
    \frametitle{Desarrollo de KPI}
    Un KPI es un indicador clave en el desempeño del negocio, el cual nos sirve para poder medir el éxito o el fracaso de nuestras acciones de marketing.
    
\end{frame}

\begin{frame}
    \frametitle{Recopilación de datos}
    
    Web server log file analysis
    \begin{itemize}
            \item Ventaja: Los datos están en los propios servidores de la empresa
            \item Ventaja: No hay llamadas a servidores externos que puedan reducir la velocidad de carga
            \item Principal problema: Caches
    \end{itemize}
    
\end{frame}

\begin{frame}
    \frametitle{Recopilación de datos}
    
    Page tagging
    \begin{itemize}
            \item Ventaja: Soluciona el problema de las páginas cache, que pueden representar hasta un tercio de todas las páginas vistas
            \item Ventaja: El page tagging puede informar información sobre eventos que no implican una solicitud al servidor web, ej: focus o clicks del mouse sobre algún boton
            \item Desventaja: Hay llamadas a servidores externos que puedan reducir la velocidad de carga
            \end{itemize}
    
\end{frame}
