    Flask
    \end{center}
    Ventajas:
    \begin{itemize}
    \item Microframework (no toma decisiones de diseño)
    \item Soporte de extensiones que agregan funcionalidad (i.e. CORS, JWTManager, Mail)
    \item Python
    \item MVT
    \end{itemize}
    

\end{frame}

\begin{frame}
    \frametitle{Back-end:}
    \begin{center}
    \includegraphics[scale=0.3]{img/flask-logo.png}\\
    Flask
    \end{center}
    Desventajas:
    \begin{itemize}
    \item Dificultad para escalar (delega decisiones arquitectonicas al desarrollador)
    \item Extensiones dependen de terceros
    \end{itemize}

\end{frame}

\begin{frame}
    \frametitle{Front-end:}
    \begin{center}
    \includegraphics[scale=0.3]{img/angular-logo.png}\\
    Angular
    \end{center}
    Ventajas:
    \begin{itemize}
    \item Typescript
    \item MVC
    \item Desarrollado y mantenido por Google
    \item Sistema de servicios inyectables (injectable service class) y dependency injection
    \item Organizado en módulos. No fuerza el uso pero lo facilita
    \item Angular Cli
    \item Event system (implementa el patrón Observer)
    \end{itemize}

\end{frame}
